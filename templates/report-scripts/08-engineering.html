// ============================================================ // 工程质量分析 // ============================================================ function renderEngineeringMetrics() { const engineering = stats.engineering; const empty = document.getElementById('engineering-empty'); const content = document.getElementById('engineering-content'); if (!engineering) { empty.innerHTML = renderSingleRepoOnlyEmptyState('工程质量分析'); empty.classList.remove('hidden'); content.classList.add('hidden'); return; } empty.classList.add('hidden'); content.classList.remove('hidden'); renderEngineeringScoreCards(engineering); renderBugHotFiles(engineering.bugFixHotFiles); renderCodeOwnership(engineering.codeOwnership); renderReviewQuality(engineering.reviewQuality); renderChangeMix(engineering.changeMix); } function renderEngineeringScoreCards(engineering) { const container = document.getElementById('engineering-score-cards'); const quality = engineering.commitQuality; const review = engineering.reviewQuality; const ownership = engineering.codeOwnership; const scoreColor = quality.score >= 80 ? 'text-emerald-500' : quality.score >= 60 ? 'text-amber-500' : 'text-rose-500'; const cards = [ ['Commit 质量', quality.score, scoreColor, '团队综合评分'], ['规范度', formatPercent(quality.conventionalRate), 'text-primary-500', 'Conventional Commits'], ['Scope 覆盖', formatPercent(quality.scopeCoverageRate), 'text-indigo-500', '带模块范围的提交'], ['平均长度', quality.averageMessageLength.toFixed(1), 'text-slate-700 dark:text-slate-200', 'Subject 字符数'], ['Review 参与', formatPercent(review.reviewParticipationRate), 'text-emerald-500', 'Merge trailer 覆盖'], ['HEAD 文件', formatNumber(ownership ? ownership.totalFiles : 0), 'text-slate-700 dark:text-slate-200', 'blame 可分析文件'], ]; container.innerHTML = `
${cards.map(([label, value, color, desc]) => `
${label}
${value}
${desc}
`).join('')}
`; } function renderBugHotFiles(bugFixHotFiles) { const container = document.getElementById('bug-hot-files-list'); const files = bugFixHotFiles.hotFiles || []; if (files.length === 0) { container.innerHTML = emptyEngineeringMessage('暂无 fix: 文件热点'); return; } container.innerHTML = `
${files.slice(0, 20).map(file => ` `).join('')}
文件路径 修复次数 作者数
${escapeHtml(file.path)} ${formatNumber(file.fixCount)} ${formatNumber(file.fixAuthors.length)}
`; } function renderCodeOwnership(ownership) { const container = document.getElementById('code-ownership-list'); const files = ownership ? ownership.files || [] : []; if (files.length === 0) { container.innerHTML = emptyEngineeringMessage('暂无 blame 所有权数据'); return; } container.innerHTML = `
${files.slice(0, 20).map(file => ` `).join('')}
文件路径 主导者 占比
${escapeHtml(file.path)} ${escapeHtml(file.ownerName)} ${formatPercent(file.ownershipRatio)}
`; } function renderReviewQuality(reviewQuality) { const container = document.getElementById('review-quality-list'); const reviewers = reviewQuality.reviewers || []; if (reviewQuality.mergeCommitCount === 0) { container.innerHTML = emptyEngineeringMessage('暂无 merge commit 数据'); return; } container.innerHTML = `
Merge
${formatNumber(reviewQuality.mergeCommitCount)}
带评审
${formatNumber(reviewQuality.reviewedMergeCount)}
参与率
${formatPercent(reviewQuality.reviewParticipationRate)}
${reviewers.length === 0 ? emptyEngineeringMessage('暂无 reviewer trailer') : reviewers.map(reviewer => `
${escapeHtml(reviewer.name)}
${escapeHtml(reviewer.email)}
${formatNumber(reviewer.commits)}
`).join('')}
`; } function renderChangeMix(changeMix) { const container = document.getElementById('change-mix-list'); const total = changeMix.createdFiles + changeMix.deletedFiles + changeMix.modifiedFiles; const featureWidth = total > 0 ? changeMix.featureRatio * 100 : 0; const refactorWidth = total > 0 ? changeMix.refactorRatio * 100 : 0; container.innerHTML = `
新增 ${formatPercent(changeMix.featureRatio)} 重构 ${formatPercent(changeMix.refactorRatio)}
${[ ['创建', changeMix.createdFiles, 'text-emerald-500'], ['删除', changeMix.deletedFiles, 'text-rose-500'], ['修改', changeMix.modifiedFiles, 'text-amber-500'], ].map(([label, value, color]) => `
${label}
${formatNumber(value)}
`).join('')}
`; } function formatPercent(value) { return (value * 100).toFixed(1) + '%'; } function emptyEngineeringMessage(message) { return `
${message}
`; }

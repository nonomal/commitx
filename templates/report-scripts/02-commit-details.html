`; }) .join(''); } function getDetailCommitKey(commit) { return `${commit.repoName || ''}::${commit.hash}`; } function renderDetailCommitList(commits) { const container = document.getElementById('detail-commits'); const meta = document.getElementById('detail-list-meta'); const pagination = document.getElementById('detail-pagination'); if (commits.length === 0) { container.innerHTML = '
